164inline Eigen::Matrix4d igl::Camera::projection() const
165{
166 Eigen::Matrix4d P;
167 using namespace std;
168 const double far = m_at_dist + m_far;
169 const double near = m_near;
170 // http://stackoverflow.com/a/3738696/148668
171 if(m_orthographic)
172 {
173 const double f = 0.5;
174 const double left = -f*m_aspect;
175 const double right = f*m_aspect;
176 const double bottom = -f;
177 const double top = f;
178 const double tx = (right+left)/(right-left);
179 const double ty = (top+bottom)/(top-bottom);
180 const double tz = (far+near)/(far-near);
181 const double z_fix = 0.5 /m_at_dist / tan(m_angle*0.5 * (igl::PI/180.) );
182 P<<
183 z_fix*2./(right-left), 0, 0, -tx,
184 0, z_fix*2./(top-bottom), 0, -ty,
185 0, 0, -z_fix*2./(far-near), -tz,
186 0, 0, 0, 1;
187 }else
188 {
189 const double yScale = tan(PI*0.5 - 0.5*m_angle*PI/180.);
190 // http://stackoverflow.com/a/14975139/148668
191 const double xScale = yScale/m_aspect;
192 P<<
193 xScale, 0, 0, 0,
194 0, yScale, 0, 0,
195 0, 0, -(far+near)/(far-near), -1,
196 0, 0, -2.*near*far/(far-near), 0;
197 P = P.transpose().eval();
198 }
199 return P;
200}

240inline Eigen::Vector3d igl::Camera::unit_plane() const
241{
242 // Distance of center pixel to eye
243 const double d = 1.0;
244 const double a = m_aspect;
245 const double theta = m_angle*PI/180.;
246 const double w =
247 2.*sqrt(-d*d/(a*a*pow(tan(0.5*theta),2.)-1.))*a*tan(0.5*theta);
248 const double h = w/a;
249 return Eigen::Vector3d(w*0.5,h*0.5,-d);
250}

269inline void igl::Camera::dolly_zoom(const double da)
270{
271 using namespace std;
272 using namespace Eigen;
273#ifndef NDEBUG
274 Vector3d old_at = at();
275#endif
276 const double old_angle = m_angle;
277 if(old_angle + da < IGL_CAMERA_MIN_ANGLE)
278 {
279 m_orthographic = true;
280 }else if(old_angle + da > IGL_CAMERA_MIN_ANGLE)
281 {
282 m_orthographic = false;
283 }
284 if(!m_orthographic)
285 {
286 m_angle += da;
287 m_angle = min(89.,max(IGL_CAMERA_MIN_ANGLE,m_angle));
288 // change in distance
289 const double s =
290 (2.*tan(old_angle/2./180.*igl::PI)) /
291 (2.*tan(m_angle/2./180.*igl::PI)) ;
292 const double old_at_dist = m_at_dist;
293 m_at_dist = old_at_dist * s;
294 dolly(Vector3d(0,0,1)*(m_at_dist - old_at_dist));
295 assert((old_at-at()).squaredNorm() < DOUBLE_EPS);
296 }

299inline void igl::Camera::turn_eye(const Eigen::Quaterniond & q)
300{
301 using namespace Eigen;
302 Vector3d old_eye = eye();
303 // eye should be fixed
304 //
305 // eye_1 = R_1 * t_1 = eye_0
306 // t_1 = R_1' * eye_0
307 m_rotation_conj = q.conjugate();
308 m_translation = m_rotation_conj * old_eye;
309 assert((old_eye - eye()).squaredNorm() < DOUBLE_EPS);
310}

312inline void igl::Camera::orbit(const Eigen::Quaterniond & q)
313{
314 using namespace Eigen;
315 Vector3d old_at = at();
316 // at should be fixed
317 //
318 // at_1 = R_1 * t_1 - R_1 * z = at_0
319 // t_1 = R_1' * (at_0 + R_1 * z)
320 m_rotation_conj = q.conjugate();
321 m_translation =
322 m_rotation_conj *
323 (old_at +
324 m_rotation_conj.conjugate() * Vector3d(0,0,1) * m_at_dist);
325 assert((old_at - at()).squaredNorm() < DOUBLE_EPS);
326}

343 Quaterniond a,b;
344 a.setFromTwoVectors(Vector3d(0,0,-1),-F);
345 b.setFromTwoVectors(a*Vector3d(0,1,0),proj_up);
346 m_rotation_conj = (b*a).conjugate();
347 m_translation = m_rotation_conj * eye;
